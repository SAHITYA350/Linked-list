#include <stdio.h>

struct slist {
    int data;
    struct slist* next;
};

typedef struct slist node;

node* start = NULL; 

void display() {
    node* tmp;
    tmp = start;
    if (tmp == NULL) {
        printf("Empty..\n");
    } else {
        printf("Display : \n");
        while (tmp != NULL) {

           
            printf("|%d|->", tmp->data);
            tmp = tmp->next;
        }
        printf("NULL");
    }
}
    void create() {

     tmp->next = NULL;

        tmp1 = start;

}
